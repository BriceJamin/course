class Maison:
    def __init__(self, t):
        self._temperature = t
        
    def get_temperature(self):
        return self._temperature
    
    def set_temperature(self, t):
        if 5 < t and t < 25:
            self._temperature = t
            return
        raise TemperatureError()
        
class TemperatureError(Exception):
    pass
